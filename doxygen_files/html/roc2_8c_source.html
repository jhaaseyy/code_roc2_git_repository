629 int SerialPortInit(serial *port)
630 {
631  struct termios options;
632 
633  printf("Opening port %s\r\n",port->port);
634  port->fp = open(port->port, O_RDWR | O_NOCTTY | O_NDELAY); // Open serial port
635  if(port->fp == BCERROR)
636  return(BCSEROPENERROR);
637 
638  tcgetattr(port->fp, &options); // Get current port options
639 
640  cfsetispeed(&options, port->speed); // Set input baud rate
641  cfsetospeed(&options, port->speed); // Set output baud rate
642 
643  options.c_cflag &= ~CSIZE; // Clear data bit fields
644  switch(port->bits)
645  {
646  case 7:
647  options.c_cflag |= CS7; // Select 7 data bits
648  break;
649  case 8:
650  options.c_cflag |= CS8; // Select 8 data bits
651  break;
652  default:
653  return(BCINVALIDBITS);
654  }
655 
656  switch(port->parity)
657  {
658  case 'N':
659  options.c_cflag &= ~PARENB; // Clear parity enable bit
660  options.c_iflag &= ~INPCK; // Disable parity checking of incoming data
661  break;
662  case 'O':
663  options.c_cflag |= PARENB; // Set parity enable bit
664  options.c_cflag |= PARODD; // Set odd parity bit
665  options.c_iflag |= (INPCK | ISTRIP); // Enable parity checking of incoming data
666  break;
667  case 'E':
668  options.c_cflag |= PARENB; // Set parity enable bit
669  options.c_cflag &= ~PARODD; // Clear odd parity bit (which means use even parity)
670  options.c_iflag |= (INPCK | ISTRIP); // Enable parity checking of incoming data
671  break;
672  default:
673  return(BCINVALIDPAR);
674  }
675 
676  switch(port->stop)
677  {
678  case 1:
679  options.c_cflag &= ~CSTOPB; // Clear stop bits bit (which means use 1 stop bit)
680  break;
681  case 2:
682  options.c_cflag |= CSTOPB; // Set stop bits bit (which means use 2 stop bits)
683  break;
684  default:
685  return(BCINVALIDSTOP);
686  }
687 
688  switch(port->flow)
689  {
690  case 'N':
691  options.c_cflag &= ~CRTSCTS; // Disable hardware flow control
692  options.c_iflag &= ~(IXON | IXOFF | IXANY); // Disable software flow control
693  break;
694  case 'H':
695  options.c_cflag |= CRTSCTS; // Enable hardware flow control
696  options.c_iflag &= ~(IXON | IXOFF | IXANY); // Disable software flow control
697  break;
698  case 'S':
699  options.c_cflag &= ~CRTSCTS; // Disable hardware flow control
700  options.c_iflag |= (IXON | IXOFF | IXANY); // Enable software flow control
701  break;
702  default:
703  return(BCINVALIDFLOW);
704  }
705 
706  options.c_iflag &= ~ICRNL; // disable mapping of CR to NL
707  options.c_lflag &= ~(ICANON | ECHO | ECHOE | ECHOK | ECHONL | ECHOCTL | ECHOPRT | ECHOKE | ISIG); // Turn off line read (canonical) mode and character echoes
708  options.c_cflag |= CLOCAL; // Set local mode
709 // options.c_cflag &= ~CREAD; // Disable the receiver for now
710  options.c_cflag |= CREAD; // Enable the receiver

1084 int SendGpsCmd(char *cmdstr, int timeout)
1085 {
1086  size_t n;
1087  struct timespec spec;
1088  int prompt, i, numread, ptr;
1089  char inbuf[256]; // needs to be big enough to hold replies from commands
1090  double nowtime, endtime;
1091 
1092 /* Get prompt to make sure GPS is ready to receive command.
1093  Return an error if prompt not received
1094 */
1095  if(verbose) printf("%s: Sending GPS command %s", MODULE_NAME, cmdstr);
1096  clock_gettime(CLOCK_MONOTONIC, &spec);
1097  nowtime = spec.tv_sec + (spec.tv_nsec / 1000000000);
1098  endtime = nowtime + (timeout/1000);
1099 
1100  write(gps_cmd.fp, "\n", 1); // send LF to get command prompt
1101  prompt=0;
1102  numread=0;
1103  inbuf[0]=0;
1104  ptr=0;
1105 
1106  while(!prompt && (nowtime < endtime)) // Wait until prompt (COMx>) is received
1107  {
1108  ioctl(gps_cmd.fp, FIONREAD, &i);
1109  if(i>0)
1110  {
1111  numread = read(gps_cmd.fp, inbuf+ptr, i);
1112  ptr += numread;
1113  inbuf[ptr] = 0;
1114  if(strstr(inbuf, "COM") && strstr(inbuf, ">"))
1115  {
1116  prompt = 1;
1117  }
1118  }
1119  clock_gettime(CLOCK_MONOTONIC, &spec);
1120  nowtime = spec.tv_sec + (spec.tv_nsec / 1000000000);
1121  }
1122  if(nowtime >= endtime) return(-1); // Timed out without receiving prompt
1123 
1124 /* Send command
1125 */
1126  n = write(gps_cmd.fp, cmdstr, strlen(cmdstr)); // Send command
1127  if(n != strlen(cmdstr)) return(-2); // Some sort of write error occurred
1128 
1129  return(BCSUCCESS);
1130
